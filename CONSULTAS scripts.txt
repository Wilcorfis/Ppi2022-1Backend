/vehiculos/maximo
SELECT * FROM VEHICULO WHERE MODELO=(
SELECT max(MODELO) FROM VEHICULO
)
/vehiculos/minimo
SELECT * FROM VEHICULO WHERE MODELO=(
SELECT min(MODELO) FROM VEHICULO
)

/vehiculos/:fecha1&:fecha2
SELECT * FROM VEHICULO WHERE FECHA_VEN_SEGURO BETWEEN '2022-01-01' AND '2022-12-12'

/vehiculos/modelos/:modelo1&:modelo2
SELECT * FROM VEHICULO WHERE MODELO BETWEEN '222' AND '666'

/vehiculos/lineasactivas
SELECT V.NUMERO_PLACA,V.MODELO, F.DESCRIPCION, M.DESCRIPCION FROM VEHICULO as V
INNER JOIN FABRICANTE AS F ON F.ID_FABRICANTE=V.FK_ID_FABRICANTE
INNER JOIN MARCA AS M ON F.FK_ID_MARCA=M.ID_MARCA
WHERE F.ACTIVO='S';

/vehiculos/sumarmodelos
SELECT SUM(MODELO) FROM VEHICULO;

/vehiculos/promediomodelos
SELECT AVG(MODELO) FROM VEHICULO;

/vehiculos/contarlineas
SELECT COUNT(ACTIVO) FROM FABRICANTE GROUP BY ACTIVO;
